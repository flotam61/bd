psql -U postgres

create table musician (id serial primary key, name varchar(32) not null, alias varchar(32) not null unique, albums varchar(32) not null, genre integer references genre(id))

create table if not exists albums ( 
	id serial primary key,
	id_musician integer references musician(id),
	name_album varchar(32) not null,
	year integer not null,
);

create table if not exists track (
	id serial primary key,
	id_album integer references albums(id),
	name_track varchar(56) not null,
	duration numeric not null
);

create table if not exists genre (
	id serial primary key,
	genre_name varchar(40)
);
